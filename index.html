function doGet(e) { // Read data for dashboard var ss = SpreadsheetApp.getActiveSpreadsheet(); var sh = ss.getSheetByName('CPC_Raw_Logs'); if (!sh || sh.getLastRow() < 2) { return ContentService .createTextOutput(JSON.stringify({logs:[]})) .setMimeType(ContentService.MimeType.JSON); } var date = e.parameter.date || ''; var site = e.parameter.site || ''; var rows = sh.getDataRange().getValues(); var headers = rows[0]; var logs = []; for (var i = 1; i < rows.length; i++) { var row = rows[i]; var obj = {}; headers.forEach(function(h, j) { obj[h] = row[j]; }); try { obj.machineData = JSON.parse(obj.machineData || '{}'); } catch(e){} try { obj.compSummary = JSON.parse(obj.compSummary || '{}'); } catch(e){} try { obj.toolChanges = JSON.parse(obj.toolChanges || '[]'); } catch(e){} if ((!date || obj.date === date) && (!site || obj.site === site)) { logs.push(obj); } } return ContentService .createTextOutput(JSON.stringify({logs:logs})) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var d = JSON.parse(e.postData.contents); // Raw log sheet — stores complete JSON for dashboard reading var raw = ss.getSheetByName('CPC_Raw_Logs') || ss.insertSheet('CPC_Raw_Logs'); if (raw.getLastRow() === 0) { raw.appendRow(['id','date','site','siteName','shift', 'supervisor','totalProduction','totalRejection', 'rejectionPct','machineData','compSummary', 'toolChanges','shiftNotes','loggedAt']); raw.getRange(1,1,1,14).setFontWeight('bold') .setBackground('#1c2936').setFontColor('#ffffff'); raw.setFrozenRows(1); } if (d.type === 'CPC_SHIFT_LOG') { raw.appendRow([ d.id, d.date, d.site, d.siteName, d.shift, d.supervisor, d.totalProduction, d.totalRejection, d.rejectionPct, JSON.stringify(d.machineData||{}), JSON.stringify(d.compSummary||{}), JSON.stringify(d.toolChanges||[]), d.shiftNotes||'', new Date().toLocaleString('en-IN') ]); logShiftSummary(ss, d); logMachineDetail(ss, d); } if (d.type === 'TOOL_CHANGE') { logToolChange(ss, d); } return ContentService .createTextOutput(JSON.stringify({status:'ok'})) .setMimeType(ContentService.MimeType.JSON); } function logShiftSummary(ss, d) { var sh = ss.getSheetByName('CPC_Shift_Log') || ss.insertSheet('CPC_Shift_Log'); if (sh.getLastRow() === 0) { sh.appendRow(['Date','Site','Shift','Supervisor', 'Total Production','Total Rejection', 'Rejection %','Tool Changes','Shift Notes','Logged At']); formatHeader(sh, 10); } sh.appendRow([d.date, d.siteName, d.shift, d.supervisor, d.totalProduction, d.totalRejection, d.rejectionPct, d.toolChanges ? d.toolChanges.length : 0, d.shiftNotes||'', new Date().toLocaleString('en-IN')]); } function logMachineDetail(ss, d) { var ms = ss.getSheetByName('CPC_Machine_Detail') || ss.insertSheet('CPC_Machine_Detail'); if (ms.getLastRow() === 0) { ms.appendRow(['Date','Site','Shift','Supervisor', 'Machine','Component','Production', 'Rejection','Rejection %','Remarks']); formatHeader(ms, 10); } var md = d.machineData || {}; Object.keys(md).forEach(function(mc) { var m = md[mc]; if (!m.production) return; var pct = m.production > 0 ? ((m.rejection||0)/m.production*100).toFixed(2) : 0; ms.appendRow([d.date, d.siteName, d.shift, d.supervisor, mc, m.component||'', m.production, m.rejection||0, pct, m.remarks||'']); }); } function logToolChange(ss, d) { var ts = ss.getSheetByName('CPC_Tool_Changes') || ss.insertSheet('CPC_Tool_Changes'); if (ts.getLastRow() === 0) { ts.appendRow(['Date','Site','Shift','Supervisor', 'Component','Machine','SAP Code','Tool Name', 'Operation','Make','Standard Life (pcs)', 'Life Achieved (pcs)','% of Standard', 'Change Reason','New Tool Fitted','Remarks','Logged At']); formatHeader(ts, 17); } if (d.toolCode) { var pct = d.stdLife > 0 ? Math.round(d.lifeAchieved/d.stdLife*100) : ''; ts.appendRow([d.date, d.siteName, d.shift, d.supervisor, d.comp, d.machine, d.toolCode, d.toolName, d.operation, d.make, d.stdLife||'', d.lifeAchieved||'', pct, d.changeReason, d.newToolFitted||'', d.remarks||'', new Date().toLocaleString('en-IN')]); return; } (d.toolChanges||[]).forEach(function(tc) { var pct = tc.stdLife > 0 ? Math.round(tc.lifeAchieved/tc.stdLife*100) : ''; ts.appendRow([d.date, d.siteName, d.shift, d.supervisor, tc.comp||'', tc.machine, tc.toolCode||'', tc.toolName||'', tc.operation||'', tc.make||'', tc.stdLife||'', tc.lifeAchieved||'', pct, tc.changeReason, tc.newToolFitted||'', tc.remarks||'', new Date().toLocaleString('en-IN')]); }); } function formatHeader(sheet, cols) { sheet.getRange(1,1,1,cols).setFontWeight('bold') .setBackground('#1c2936').setFontColor('#ffffff'); sheet.setFrozenRows(1); }
